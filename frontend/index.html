Internet Identity Auth Login

Login with Internet Identity authentication

<button class="auth-button" onclick="signInWithInternetIdentity()">
    <img src="ii-logo.png" alt="Internet Identity logo" />
    Sign in with Internet Identity
</button>

<script>
async function signInWithInternetIdentity() {
    const authClient = await AuthClient.create();
    const isAuthenticated = await authClient.isAuthenticated();

    if (!isAuthenticated) {
        await authClient.login({
            identityProvider: "https://identity.ic0.app/#authorize",
            onSuccess: () => {
                // Handle successful login
            }
        });
    }
}
</script>
